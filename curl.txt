# don't display progress, but still show errors and exit non-0 when failed:
curl --silent --show-error --fail --fail-early http://foo.bar/stuff
# the same, but shorter:
curl -sSf --fail-early http://foo.bar/stuff
# follow all redirections
curl -L baszerr.eu

# common use case to make curl useful in scripts:
curl -fsSL baszerr.eu
